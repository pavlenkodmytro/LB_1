#include <iostream>  // Підключення бібліотеки для введення та виведення даних (std::cin і std::cout).
#include <cmath>     // Підключення бібліотеки для використання математичних функцій, таких як exp(), sin(), pow(), M_PI (константа π).

int main() 
{
    // Оголошення змінних типу double для зберігання значень, введених користувачем.
    double a, b;

    // Виведення на екран запиту на введення значення для змінної 'a'.
    std::cout << "Enter a: ";  
    // Читання введеного значення з клавіатури та його збереження у змінну 'a'.
    std::cin >> a;

    // Виведення на екран запиту на введення значення для змінної 'b'.
    std::cout << "Enter b: ";  
    // Читання введеного значення з клавіатури та його збереження у змінну 'b'.
    std::cin >> b;

    // Перевірка, чи не є 'a' нульовим. Це необхідно, щоб уникнути ділення на нуль у знаменнику.
    if (a == 0) {
        // Якщо 'a' дорівнює нулю, виводимо повідомлення про помилку і завершуємо виконання.
        std::cout << "Error: 'a' cannot be zero!" << std::endl;  
    } 
    else {
        // 1. ЧИСЕЛЬНИК:
        // Обчислюємо першу частину чисельника: 0.75 * e^(1 - b)
        // Використовуємо функцію exp() для обчислення експоненти: e^(1 - b), після чого множимо на 0.75.
        double numerator_part1 = 0.75 * exp(1 - b);

        // Обчислюємо другу частину чисельника: 0.31 * e^(1 - a)
        // Використовуємо функцію exp() для обчислення експоненти: e^(1 - a), після чого множимо на 0.31.
        double numerator_part2 = 0.31 * exp(1 - a);

        // Обчислюємо загальний чисельник як суму двох частин: 
        // 0.75 * e^(1 - b) + 0.31 * e^(1 - a)
        double numerator = numerator_part1 + numerator_part2;

        // 2. ЗНАМЕННИК:
        // Обчислюємо аргумент для синуса: b / a * π
        // Ми ділимо 'b' на 'a', після чого множимо результат на константу π (M_PI).
        double sine_argument = (b / a) * M_PI;

        // Обчислюємо значення синуса від цього аргументу: sin(b / a * π)
        // Використовуємо функцію sin() для обчислення синуса.
        double sine_value = sin(sine_argument);

        // Підносимо результат синуса до квадрату: sin²(b / a * π)
        // Використовуємо функцію pow() для піднесення значення синуса до квадрату.
        double sine_squared = pow(sine_value, 2);

        // Обчислюємо загальний знаменник: 0.731 + sin²(b / a * π)
        double denominator = 0.731 + sine_squared;

        // 3. РЕЗУЛЬТАТ:
        // Обчислюємо остаточний результат як частку чисельника і знаменника.
        // Виконується ділення чисельника на знаменник.
        double result = numerator / denominator;

        // Виведення результату обчислень на екран.
        std::cout << "Result: " << result << std::endl;
    }

    // Повертаємо 0, що вказує на успішне завершення програми.
    return 0;
}
